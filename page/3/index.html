题目

设计一个Fraction类的结构，该类表示一个可以单步前进的广义整数。该类有以下两个私有属性
int den; //单步前进的步进距离,也就是执行++运算后的增长量；
int num; //当前的整数值
设计和实现该类相应的结构函数，print函数，并重载它的前缀++运算符号和后缀++运算符号。
该类完成后，如下main()函数可以执行。
void main()
{
Fraction b(1,10),a(4,-8);
(a++).print();
a.print();
a = b++;
a.print();
a = ++b;
a.print();
}
执行的结果为
-8
-4
10
12

分析

这道题主要考察的是前置++和后置++运算符的重载

Fraction& operator++(); // 前置++
Fraction operator++(int); // 后置++

后置++运算符必须要在后面加一个int形参，表示与前置++的区别，同时，由于是++运算，所以可以返回自身。前置++运算最好要返回引用，后置可以直接返回指针。

代码

#include

using namespace std;

class Fraction{
private:
int den; //单步前进的步进距离,也就是执行++运算后的增长量；
int num; //当前的整数值

public:
Fraction(int den, int num);
Fraction& operator++(); // 前置++
Fraction operator++(int); // 后置++
void print();
};

Fraction::Fraction(int den, int num) {
this->den = den;
this->num = num;
}

Fraction & Fraction::operator++() {
/*
* 前置++运算符
* 先++，再参与运算
* 在++之后，直接返回自身
*
* */
num += den;
return *this;
}

Fraction Fraction::operator++(int) {
/*
* 后置++运算符
* 先参与运算再++
* 先用一个temp保存要返回的值
* 在自身++
*
* */
Fraction temp(*this);
num += den;
return temp;
}

void Fraction::print() {
cout << num << endl;
}

int main() {
Fraction b(1,10),a(4,-8);

(a++).print();
a.print();

a = b++;
a.print();

a = ++b;
a.print();
return 0;

}

题目

设计一个Complex类的结构，改类表示复数。并设计和实现该类相应的结构函数，show函数，并重载它的+和*运算符号。
该类完成后，如下main()函数可以执行。
void main()
{
Complex c1(3, 5);
Complex c2 = 4;
c1 = c1 + c2;
c2 = c1 * c2;
c1.show();
c2.show();
}
执行的结果为
7+5i
28+20i

分析

重载符号结构

重载符号运算符的结构要记好，函数的名字叫做operator，后面加上要重载的符号。最好要传入引用，但是做运算不能叠加在自己的身上，要返回一个新的对象。

代码

#include

using namespace std;

class Complex{
private:
int real;
int image;
public:
Complex(int real, int image);
Complex(int real);
Complex operator+ (Complex& c); // 重载+号运算符
Complex operator* (Complex& c); // 重载*号运算符
void show();
};

Complex::Complex(int real, int image) {
this->real = real;
this->image = image;
}

Complex::Complex(int real) {
this->real = real;
this->image = 0;
}

// 注意，在下面两个重载中，不能返回自身的引用，要重新返回一个新的对象
Complex Complex::operator+(Complex &c) {
return Complex(this->real + c.real, this->image + c.image);

/\*
 \* 不能采用下面这种方法
 \* 因为返回自身的的引用的话，运算的所有基础都要叠加在第一个运算数上面
 \* 比如说：
 \* c1 = c1 + c2
 \* 这种情况下计算的结果会叠加在c1上，并且返回c1
 \* c2 = c1 + c2
 \* 这种情况计算结果也是会叠加在c1上面，并且返回c1在c2上，
 \* 但是实际上c1的值应该不需要改变
 \*
 \* \*/

// this->real += c.real;
// this->image += c.image;
// return *this;
}

Complex Complex::operator*(Complex &c) {
return Complex(this->real*c.real - this->image*c.image, this->image*c.real + this->real*c.image);
// 不能采用这种方法
// this->real = this->real*c.real - this->image*c.image;
// this->image = this->image*c.real + this->real*c.image;
// return *this;
}

void Complex::show() {
if (this->real == 0) {
cout << this->image << ‘i’ << endl;
} else if (this->image == 0){
cout << this->real << endl;
} else{
cout << this->real << ‘+’ << this->image << ‘i’ << endl;
}
}

int main() {
Complex c1(3, 5);
Complex c2 = 4;
c1 = c1 + c2;
c2 = c1 * c2;
c1.show();
c2.show();
return 0;
}

题目

设计一个名为Arrayint的类的，该类的两个私有属性分别为
int *p; //指向存放整个数组的数据区。
int n; //整个数组的整数个数
p所指向的数据区在结构函数中用动态分配，并在析构函数中释放。
该类可以实现深拷贝。设计和实现该类相应的深拷贝函数和show成员函数，show函显示数组中所有数据。
该类完成后，如下main()函数可以执行。
void main()
{
int a[10] = {5,7,8,9,10,11,12,34,32,22};
Arrayint *x = new Arrayint(10, a);
a[2] = 100;
Arrayint y = *x;
x->show();
delete x;
y.show();
}
执行的结果为:
5 7 8 9 10 11 12 34 32 22
5 7 8 9 10 11 12 34 32 22

分析

这道题主要考察深拷贝的内容。在深拷贝的时候一定要给对象中的指针分配独自的内存空间，比如说下面进行数组的深拷贝。

int *p = new a[10] // 这边就是对p指针进行分配十个int大小的内存空间分配

代码

#include

using namespace std;

class Arrayint{
private:
int *p; //指向存放整个数组的数据区。
int n; //整个数组的整数个数
public:
Arrayint(int n, int *a);
void show();
};

Arrayint::Arrayint(int n, int a[]) {
int i = 0;
this->n = n;
p = new int[n]; // 实现深拷贝，给p分配内存空间
for(i = 0; i < n; i++){
p[i] = a[i];
}
}

void Arrayint::show() {
int i;
for(i = 0; i < n; i++){
cout << p[i] << ‘ ‘;
}
cout << endl;
}

int main() {
int a[10] = {5,7,8,9,10,11,12,34,32,22};

Arrayint *x = new Arrayint(10, a);
a[2] = 100;
Arrayint y = *x;
x->show();
delete x;
y.show();
return 0;
}

题目

设计一个stack类的结构，该类可以完成栈的功能。该栈的元素类型为int，最大深度为100。它能完成push和pop方法，功能分别为压栈和出栈。
如下main()函数可以执行。
void main()
{
stack s1;
int i;
for(i = 1; i < 10;i++)
s1.push(i);
for(i = 1; i <6;i++)
cout << s1.pop()<<”,”;
s1.push(200);
cout << s1.pop()<<endl;
}
执行的结果为
9,8,7,6,5,200

分析

第二题也没有什么难度，主要要注意一下栈的溢出和触底的检验操作

代码

#include

#define maxN 100 // 栈的最大容量

using namespace std;

class stack{
private:
int top; // 栈顶指针，初始化的时候指向-1
int a[maxN]; // 用数组来模拟栈
public:
stack(); // 空参构造器
void push(int num); // push方法
int pop(); // pop方法
};

// 初始化栈
stack::stack() {
top = -1;
}

void stack::push(int num) {
if (top < maxN-1) { // 检验栈有没有溢出
top++;
a[top] = num;
}
}

int stack::pop() {
if (top > -1) { // 检验栈有没有触底
top–;
return a[top+1];
}
return -1;
}

int main()
{
stack s1;
int i;

for(i = 1; i < 10;i++)
    s1.push(i);
for(i = 1; i <6;i++)
    cout << s1.pop()<<",";
s1.push(200);
cout << s1.pop()<<endl;
return 0;

}

题目

设计一个Complex类的结构，该类表示复数。并设计和实现该类相应的结构函数和add成员函数，该函数完成复数的加法。该类完成后，如下main()函数可以执行。
void main()
{
Complex c1(3, 5);
Complex c2 = 4;
c1.add(c2);
c1.show();
c2.show();
Complex c3(0,5);
c3.show();
}
执行的结果为
7+5i
4
5i

分析

这道题没啥难度，只要是考察构造函数的使用还有类内部方法的设计。

代码

#include

using namespace std;

class Complex{
private:
int real;
int image;
public:
// 构造器
Complex(int real);
Complex(int real, int image);
void add(Complex c);
void show();
};

// 构造器和方法的具体实现
Complex::Complex(int real) {
this->real = real;
this->image = 0;
}

Complex::Complex(int real, int image) {
this->real = real;
this->image = image;
}

void Complex::add(Complex c) {
this->real += c.real;
this->image += c.image;
}

void Complex::show() {
if (real == 0) {
cout << image << ‘i’ << endl;
} else if (image == 0) {
cout << real << endl;
} else{
cout << real << ‘+’ << image << ‘i’ << endl;
}
}

int main()
{
Complex c1(3, 5); // 使用第二个构造器
Complex c2 = 4; // 使用第一个构造器
c1.add(c2); // 传入Complex对象，调用add方法
c1.show(); // 空参，调用add方法
c2.show(); // 同上
Complex c3(0,5); // 调用第二个构造器
c3.show(); // 空参，调用add方法
return 0;
}

SingleSelectForXXX
